/// Transport the ClientHello was carried over — sets the JA4 protocol letter (FoxIO spec).
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum Ja4Transport {
    Tcp,
    Quic,
}

impl Ja4Transport {
    fn marker(self) -> char {
        match self {
            Ja4Transport::Tcp => 't',
            Ja4Transport::Quic => 'q',
        }
    }
}
